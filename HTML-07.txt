FAQ's:

5. How to define one set of statements for multiple cases?
A. By defining the cases line-by-line

	case 1:
	case 2:
	case 3:
		statements;
		break;

eg.

<script>
    function f1() {
        var product = prompt("Enter Product Name");
        switch (product) {
            case "TV":
            case "Mobile":
            case "Watch":
                document.write(`${product} belongs to Electronics category`);
                break;
            case "Casuals":
            case "Boots":
                document.write(`${product} belongs to Footwear category`);
                break;
            default:
                document.write(`Avail Products : TV, Mobile, Watch, Casuals, Boots only..`);
                break;
        }
    }
    f1()
</script>


- If you case is refering to one character then you can write multiple cases like for Y and N : Y,y,n,N etc
- but for string we use to lowercase()

eg.

<script>
    function f1() {
        var choice = prompt("Enter Choice Yes or No");
        switch (choice.toLowerCase()) {
            case "yes":
                document.write("You selected yes to continue");
                break;
            case "no":
                document.write("You selected No to stop");
                break;
            default:
                document.write("Please enter your choice Y or N");
                break;
        }
    }
    f1();
</script>

6. How you write case for range of values?
A. By using Boolean expression that verifies the range of values.
   IF case is using a Boolean expression then the switch must always use true as value.

Syntax:

	switch(true)
	{
		case booleanExpression:
			statements;
			break;
	}
	
- When we are writing cases with boolean value or expression , switch value must be true eg.switch(true)

eg.

<script>
    function f1() {
        var n = parseInt(prompt("Enter Number"));
        switch (true) {
            case (n >= 1 && n <= 10):
                document.write(`Your number ${n} is between 1 to 10`);
                break;
            case (n >= 11 && n <= 20):
                document.write(` Your number ${n} is between 11 to 20 `);
                break;
            default:
                document.write("Pleae enter a number between 1 to 20 only");
                break;
        }
    }
    f1();
</script>

Adding Elements into Array:

	push()		: It adds new element(s) as last item.
	unshift()	: It adds new element(s) as first item.
	splice()	: It adds new element(s) at specific position.

Syntax:

	collection.push(e1, e2, ... )
	collection.unshift(e1, e2, ... )
	collection.splice(indexNumber, deleteCount, ...Items)

Note: Set delete count to "0" to add element without deleting existing.

Ex:

	menu.splice(1, 0, "About");
	menu.push("About, "Contact");

Removing Elements from Array:

	pop()		: It removes and returns last element.
	shift()		: It removes and returns first element.
	splice()	: It removes and returns specific element(s).

Syntax:

	collection.pop()
	collection.shift()
	collection.splice(1,2);		// from 1 index remove 2 items

            // menuItems.push(item_name);
            // alert(`${item_name} Added Successfully.. `);
            // LoadMenuItems();
            // document.getElementById("txtName").value = "";
            // //After adding in inputbox we need to clean up

            var item_name = document.getElementById("txtName").value;
            if (menuItems.indexOf(item_name) == -1) {
                menuItems.push(item_name);
                alert(`${item_name} Added Successfully.. `);
                LoadMenuItems();
                document.getElementById("txtName").value = "";
            } else {
                alert(`${item_name} Exists`);
            }
        }


        function RemoveClick() {
            var selectedItem = document.getElementById("lstMenuItems").value;
            var selectedItemIndex = menuItems.indexOf(selectedItem);
            var flag = confirm(`Are you sure ?\nWant to delete ${selectedItem}`);
            if (flag == true) {
                menuItems.splice(selectedItemIndex, 1);
                LoadMenuItems();
            }
        }

        function RemoveAllclick() {
            var flag = confirm(`Are you sure ?\nWant to delete All item`);
            if (flag) {
                menuItems.length = 0;
                LoadMenuItems();
            }
        }

        function EditClick() {
            var selectedItem = document.getElementById("lstMenuItems").value;
            document.getElementById("txtEdit").value = selectedItem;
        }

        function SaveClick() {
            var newValue = document.getElementById("txtEdit").value;
            var selectedItem = document.getElementById("lstMenuItems").value;
            var selectedIndex = menuItems.indexOf(selectedItem);

            menuItems[selectedIndex] = newValue;
            LoadMenuItems();
        } 

        function SortAsc() {
            menuItems.sort();
            LoadMenuItems();
        }
        function SortDesc() {
            menuItems.sort();
            menuItems.reverse();
            LoadMenuItems();

        }

    </script>
